2️⃣ Java Notification Rule

Ye rule manager ko notification bhejta hai.

import sailpoint.api.SailPointContext;
import sailpoint.object.EmailTemplate;
import sailpoint.object.EmailOptions;
import sailpoint.object.Identity;

public class ManagerNotificationRule {

    public static void notifyManager(
            SailPointContext context,
            Identity identity) throws Exception {

        EmailTemplate template =
                context.getObject(
                        EmailTemplate.class,
                        "Manager Approval Notification");

        String managerEmail =
                identity.getManager().getEmail();

        EmailOptions options =
                new EmailOptions(
                        managerEmail,
                        null);

        context.sendEmailNotification(
                template,
                options);

        System.out.println(
        "Notification sent to Manager: " + managerEmail);

    }
}
