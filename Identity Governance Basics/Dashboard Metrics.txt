3️⃣ Context Based Dashboard Metrics Project:-
import sailpoint.api.SailPointContext;
import sailpoint.object.Identity;
import sailpoint.object.QueryOptions;
import sailpoint.tools.GeneralException;

import java.util.Iterator;

public class DashboardMetricsService {

    public void generateMetrics(SailPointContext context) throws GeneralException {

        QueryOptions qo = new QueryOptions();

        Iterator<Identity> identities = context.search(Identity.class, qo);

        int totalUsers = 0;
        int adminUsers = 0;
        int highRiskUsers = 0;
        int violations = 0;

        while (identities.hasNext()) {

            Identity id = identities.next();

            totalUsers++;

            String roles = String.valueOf(id.getAssignedRoles());

            int riskScore = 0;

            if (roles.contains("Admin")) {
                adminUsers++;
                riskScore += 30;
            }

            if (roles.contains("DatabaseAdmin")) {
                riskScore += 40;
            }

            if (roles.contains("Finance") && roles.contains("Audit")) {
                violations++;
                riskScore += 50;
            }

            if (riskScore >= 60) {
                highRiskUsers++;
            }
        }

        System.out.println("===== DASHBOARD METRICS =====");

        System.out.println("Total Users : " + totalUsers);
        System.out.println("Admin Users : " + adminUsers);
        System.out.println("High Risk Users : " + highRiskUsers);
        System.out.println("Policy Violations : " + violations);

    }
}
4️⃣ Rule / Task jo dashboard metrics run karega:-
import sailpoint.api.SailPointContext;

public Object execute(SailPointContext context) throws Exception {

    DashboardMetricsService service = new DashboardMetricsService();

    service.generateMetrics(context);

    return "Dashboard Metrics Generated";

}
