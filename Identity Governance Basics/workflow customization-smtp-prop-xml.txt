2️⃣ Identity Creation (SailPoint style):-
import sailpoint.object.Identity;
import sailpoint.api.SailPointContext;

public class IdentityCreator {

    public Identity createIdentity(SailPointContext context,
                                   String empId,
                                   String name,
                                   String email) throws Exception {

        Identity identity = new Identity();

        identity.setName(empId);
        identity.setDisplayName(name);
        identity.setEmail(email);

        context.saveObject(identity);
        context.commitTransaction();

        return identity;
    }
}
3️⃣ Role Assignment Rule:-
import sailpoint.object.Identity;
import sailpoint.object.Bundle;
import sailpoint.api.SailPointContext;

public class RoleAssignmentRule {

    public void assignEmployeeRole(SailPointContext context,
                                   Identity identity) throws Exception {

        Bundle role = context.getObject(Bundle.class, "EmployeeRole");

        identity.addAssignedRole(role);

        context.saveObject(identity);
        context.commitTransaction();
    }
}

Result
➡ User ko EmployeeRole mil gaya.

5️⃣ AD Account Provisioning Rule:-
import sailpoint.object.ProvisioningPlan;
import sailpoint.object.AccountRequest;
import sailpoint.object.Identity;

public class ADProvisioningRule {

    public ProvisioningPlan createADAccount(Identity identity) {

        ProvisioningPlan plan = new ProvisioningPlan();
        plan.setIdentity(identity);

        AccountRequest accReq = new AccountRequest();

        accReq.setApplication("Active Directory");
        accReq.setNativeIdentity(identity.getName());
        accReq.setOperation(AccountRequest.Operation.Create);

        plan.add(accReq);

        return plan;
    }
}
Result
➡ AD account create request provisioning engine ko milega.

6️⃣ Real Email Send (SMTP Example):-

Yeh JavaMail SMTP example hai.

import java.util.Properties;

import javax.mail.Message;
import javax.mail.PasswordAuthentication;
import javax.mail.Session;
import javax.mail.Transport;

import javax.mail.internet.InternetAddress;
import javax.mail.internet.MimeMessage;

public class WelcomeEmailSender {

    public static void sendEmail(String toEmail) {

        final String fromEmail = "admin@company.com";
        final String password = "smtp-password";

        Properties props = new Properties();

        props.put("mail.smtp.host", "smtp.gmail.com");
        props.put("mail.smtp.port", "587");
        props.put("mail.smtp.auth", "true");
        props.put("mail.smtp.starttls.enable", "true");

        Session session = Session.getInstance(props,
            new Authenticator() {
                protected PasswordAuthentication getPasswordAuthentication() {
                    return new PasswordAuthentication(fromEmail, password);
                }
            });

        try {

            Message message = new MimeMessage(session);

            message.setFrom(new InternetAddress(fromEmail));
            message.setRecipients(Message.RecipientType.TO,
                    InternetAddress.parse(toEmail));

            message.setSubject("Welcome to Company");

            message.setText(
                    "Hello,\n\nYour company account has been created successfully.\n\nRegards\nIT Team");

            Transport.send(message);

            System.out.println("Email Sent Successfully");

        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}

<?xml version="1.0" encoding="UTF-8"?>

<Workflow name="JoinerWorkflow"
          type="Identity"
          explicitTransitions="true">

    <Variable name="empId"/>
    <Variable name="name"/>
    <Variable name="email"/>

    <!-- START -->
    <Step name="Start">
        <Transition to="Fetch HR Data"/>
    </Step>

    <!-- HR DATA -->
    <Step name="Fetch HR Data">

        <Script>
            <Source>

            java.util.Map emp = HRSystem.getEmployeeData();

            workflow.put("empId", emp.get("empId"));
            workflow.put("name", emp.get("name"));
            workflow.put("email", emp.get("email"));

            </Source>
        </Script>

        <Transition to="Create Identity"/>
    </Step>

    <!-- CREATE IDENTITY -->
    <Step name="Create Identity">

        <Script>
            <Source>

            IdentityCreator creator = new IdentityCreator();

            sailpoint.object.Identity identity =
                    creator.createIdentity(context,
                    (String)workflow.get("empId"),
                    (String)workflow.get("name"),
                    (String)workflow.get("email"));

            workflow.put("identity", identity);

            </Source>
        </Script>

        <Transition to="Assign Role"/>
    </Step>

    <!-- ROLE ASSIGN -->
    <Step name="Assign Role">

        <Script>
            <Source>

            RoleAssignmentRule roleRule =
                    new RoleAssignmentRule();

            roleRule.assignEmployeeRole(context,
                    (sailpoint.object.Identity)workflow.get("identity"));

            </Source>
        </Script>

        <Transition to="Manager Approval"/>
    </Step>

    <!-- APPROVAL -->
    <Step name="Manager Approval">

        <Approval owner="identity.manager"
                  description="Manager Approval for Joiner"/>

        <Transition to="Provision AD Account"/>
    </Step>

    <!-- PROVISION ACCOUNT -->
    <Step name="Provision AD Account">

        <Script>
            <Source>

            ADProvisioningRule rule =
                    new ADProvisioningRule();

            rule.createADAccount(
                    (sailpoint.object.Identity)workflow.get("identity"));

            </Source>
        </Script>

        <Transition to="Send Welcome Email"/>
    </Step>

    <!-- EMAIL -->
    <Step name="Send Welcome Email">

        <Script>
            <Source>

            String email = (String)workflow.get("email");

            WelcomeEmailSender.sendEmail(email);

            </Source>
        </Script>

        <Transition to="End"/>
    </Step>

    <Step name="End">
        <End/>
    </Step>

</Workflow>
