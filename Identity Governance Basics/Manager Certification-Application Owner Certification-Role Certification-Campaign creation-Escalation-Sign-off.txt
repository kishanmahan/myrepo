1️⃣ Access Review / Certification kya hai?
Access Review / Certification ka matlab:-
➡ Ye ek process hai jisme manager ya reviewer check karta hai ki user ke access sahi hai ya nahi.
Example: Rahul ko FinanceRole mila hai
Manager check karega: “Kya Rahul ko ye access abhi bhi chahiye?”
Agar nahi chahiye → revoke kar denge
Agar chahiye → approve kar denge.

2️⃣ Manager Certification:-
Manager ke liye specially SailPoint UI me task generate hota hai
Manager review karta hai sab team members ke access
Approve / Revoke kar sakta hai.

3️⃣ Automated Flow:-
SailPoint automatically Access Review campaigns generate karta hai
Manager ko notify karta hai
Manager login karke Approve / Revoke karta hai
Provisioning engine automatically access update kar deta hai.

1️⃣ Campaign kya hai?
Kya hai:
Campaign basically ek review/ certification process ka container hai.
Matlab, jab company ko decide karna hota hai ki “kaun kaun se users ke kaun kaun se access sahi hai
aur kaun se remove karne chahiye”, to us process ko ek Campaign ke andar organize kiya jata hai.
Kya karti hai:-
Kaun se users ko review karna hai (scope)
Kaun se access ya entitlements check karne hai
Review ka start aur end date define karna
Kaun kaun reviewers involved hain (manager, application owner, etc.)
→ Basically Campaign ek review session ka master object hai.

2️⃣ ReviewTask kya hai?
Kya hai:-
Jab Campaign start hota hai, to SailPoint automatically ReviewTask create karta hai.
Ye task ek specific reviewer ko assign hota hai.
Kya karti hai:-
Ek reviewer ko bolti hai “tu in users ke access check kar, approve ya revoke kar.”
Task ke andar ReviewItems hote hain, matlab actual permissions ya entitlements jo review karni hai.
Track karta hai ki task complete hua ya pending hai.

3️⃣ ReviewItem kya hai?
Kya hai:-
ReviewItem ek specific access/ entitlement ka review unit hai.
Matlab, ye smallest element hai jise reviewer approve ya revoke karta hai.
Kya karti hai:-
Example: User “John” ko “FinanceGroup” access hai → ye ek ReviewItem banega.
Reviewer ke paas options hote hain: Approve / Revoke / Do Nothing / Delegate.
Status update karta hai ki ye access ok hai ya remove karna hai.

import sailpoint.api.SailPointContext;
import sailpoint.object.Campaign;
import sailpoint.object.ReviewTask;
import sailpoint.object.ReviewItem;
import sailpoint.object.Identity;
import sailpoint.object.Entitlement;
import java.util.List;
import java.util.ArrayList;
import java.util.Date;

public class AccessReviewExample {

    // 1️⃣ Campaign Create
    public Campaign createCampaign(SailPointContext context, String name, Date startDate, Date endDate) throws Exception {
        Campaign campaign = new Campaign();
        campaign.setName(name);
        campaign.setDescription("Access review campaign");
        campaign.setStartDate(startDate);
        campaign.setEndDate(endDate);
        context.saveObject(campaign);
        context.commitTransaction();
        System.out.println("Campaign created: " + campaign.getName());
        return campaign;
    }

    // 2️⃣ ReviewTask Create
    public ReviewTask createReviewTask(SailPointContext context, Campaign campaign, Identity reviewer, List<Identity> users) throws Exception {
        ReviewTask task = new ReviewTask();
        task.setCampaign(campaign);
        task.setReviewer(reviewer);
        task.setDescription("Review task for " + reviewer.getDisplayName());

        List<ReviewItem> items = new ArrayList<>();
        for (Identity user : users) {
            for (Entitlement ent : user.getEntitlements()) {
                ReviewItem item = new ReviewItem();
                item.setIdentity(user);
                item.setEntitlement(ent);
                item.setStatus("Pending");
                items.add(item);
            }
        }

        task.setItems(items);
        context.saveObject(task);
        context.commitTransaction();
        System.out.println("ReviewTask created for: " + reviewer.getDisplayName());
        return task;
    }

    // 3️⃣ ReviewItem Status Update
    public void updateReviewItemStatus(SailPointContext context, ReviewItem item, String action) throws Exception {
        if(action.equalsIgnoreCase("Approve")) item.setStatus("Approved");
        else if(action.equalsIgnoreCase("Revoke")) item.setStatus("Revoked");
        else item.setStatus("Pending");

        context.saveObject(item);
        context.commitTransaction();
        System.out.println("ReviewItem updated: " + item.getIdentity().getDisplayName() + " - " + item.getEntitlement().getName() + " = " + item.getStatus());
    }

    // 4️⃣ Full Flow
    public void runAccessReview(SailPointContext context) throws Exception {
        // Campaign
        Campaign campaign = createCampaign(context, "Finance Access Review", new Date(), new Date(System.currentTimeMillis() + 7*24*60*60*1000));

        // Reviewer & Users
        Identity manager = context.getObjectByName(Identity.class, "manager.john");
        List<Identity> users = new ArrayList<>();
        users.add(context.getObjectByName(Identity.class, "user.alice"));
        users.add(context.getObjectByName(Identity.class, "user.bob"));

        // ReviewTask
        ReviewTask task = createReviewTask(context, campaign, manager, users);

        // Approve/Revoke Logic
        for(ReviewItem item : task.getItems()){
            if(item.getEntitlement().getName().equals("FinanceGroup"))
                updateReviewItemStatus(context, item, "Revoke");
            else
                updateReviewItemStatus(context, item, "Approve");
        }

        System.out.println("Access review campaign completed: " + campaign.getName());
    }
}

import sailpoint.api.SailPointContext;
import sailpoint.object.Campaign;
import sailpoint.object.Identity;
import sailpoint.object.Application;
import sailpoint.object.ReviewTask;
import sailpoint.object.ReviewItem;
import sailpoint.object.QueryOptions;
import sailpoint.object.Filter;
import sailpoint.tools.GeneralException;
import java.util.List;
import java.util.Iterator;
public class ApplicationOwnerCertificationProcessor {

    private SailPointContext context;

    public ApplicationOwnerCertificationProcessor(SailPointContext context) {
        this.context = context;
    }

    public void processCertification(String campaignName) throws GeneralException {

        Campaign campaign = context.getObjectByName(Campaign.class, campaignName);

        if (campaign == null) {
            System.out.println("Campaign not found");
            return;
        }

        List tasks = campaign.getTasks();

        for (Object obj : tasks) {

            ReviewTask task = (ReviewTask) obj;

            Identity reviewer = task.getReviewer();

            System.out.println("Reviewer: " + reviewer.getName());

            List items = task.getItems();

            for (Object itemObj : items) {

                ReviewItem item = (ReviewItem) itemObj;

                Identity identity = item.getIdentity();

                String entitlement = item.getEntitlement();

                Application app = item.getApplication();

                System.out.println("User: " + identity.getName());
                System.out.println("Application: " + app.getName());
                System.out.println("Entitlement: " + entitlement);

                if ("SAP_ADMIN_ROLE".equals(entitlement)) {

                    item.setDecision("Revoke");
                    item.setComment("Admin role not required");

                } else {

                    item.setDecision("Approve");
                    item.setComment("Access valid");

                }

                context.saveObject(item);

            }

        }

        context.commitTransaction();
    }

}

import sailpoint.api.SailPointContext;
import sailpoint.object.ReviewItem;
import sailpoint.object.ProvisioningPlan;
import sailpoint.object.AccountRequest;
import sailpoint.object.AttributeRequest;
import sailpoint.object.Identity;
import sailpoint.tools.GeneralException;
public void processReviewItem(
        SailPointContext context,
        ReviewItem item
) throws GeneralException {

    String decision = item.getDecision();

    if ("Revoke".equals(decision)) {

        Identity identity = item.getIdentity();

        String entitlement = item.getEntitlement();

        ProvisioningPlan plan = new ProvisioningPlan();

        plan.setIdentity(identity);

        AccountRequest accountRequest = new AccountRequest();

        accountRequest.setOperation(AccountRequest.Operation.Modify);

        AttributeRequest attrRequest = new AttributeRequest();

        attrRequest.setName("memberOf");

        attrRequest.setOperation(AttributeRequest.Operation.Remove);

        attrRequest.setValue(entitlement);

        accountRequest.add(attrRequest);

        plan.add(accountRequest);

        System.out.println("Access removed for: " + identity.getName());

    }
}

import sailpoint.api.SailPointContext;
import sailpoint.object.Campaign;
import sailpoint.object.ReviewTask;
import sailpoint.object.Identity;
import sailpoint.tools.GeneralException;

import java.util.Date;
import java.util.List;
public class CertificationEscalationProcessor {

    private SailPointContext context;

    public CertificationEscalationProcessor(SailPointContext context) {
        this.context = context;
    }

    public void processCampaignEscalation(String campaignName)
            throws GeneralException {

        Campaign campaign =
                context.getObjectByName(Campaign.class, campaignName);

        if (campaign == null) {
            System.out.println("Campaign not found");
            return;
        }

        List tasks = campaign.getTasks();

        for (Object obj : tasks) {

            ReviewTask task = (ReviewTask) obj;

            if (isTaskExpired(task)) {

                escalateTask(task);

            }

        }

        context.commitTransaction();
    }
private boolean isTaskExpired(ReviewTask task) {

    Date dueDate = task.getDueDate();

    if (dueDate == null) {
        return false;
    }

    Date currentDate = new Date();

    return currentDate.after(dueDate);
}
private void escalateTask(ReviewTask task)
        throws GeneralException {

    Identity currentReviewer = task.getOwner();

    if (currentReviewer == null) {
        return;
    }

    Identity manager = currentReviewer.getManager();

    if (manager == null) {
        System.out.println("No manager found for escalation");
        return;
    }

    task.setOwner(manager);

    context.saveObject(task);

    System.out.println(
        "Task escalated from "
        + currentReviewer.getName()
        + " to "
        + manager.getName()
    );
}
}

import sailpoint.api.SailPointContext;
import sailpoint.object.Campaign;
import sailpoint.object.ReviewTask;
import sailpoint.object.ReviewItem;
import sailpoint.tools.GeneralException;

import java.util.List;

public class CertificationSignOffProcessor {

    private SailPointContext context;

    public CertificationSignOffProcessor(SailPointContext context) {
        this.context = context;
    }

    public void processSignOff(String campaignName)
            throws GeneralException {

        Campaign campaign =
                context.getObjectByName(Campaign.class, campaignName);

        if (campaign == null) {
            System.out.println("Campaign not found");
            return;
        }

        List tasks = campaign.getTasks();

        for (Object obj : tasks) {

            ReviewTask task = (ReviewTask) obj;

            processTask(task);
        }

        campaign.setStatus("Completed");

        context.saveObject(campaign);

        context.commitTransaction();

        System.out.println("Campaign Signed Off Successfully");
    }

    private void processTask(ReviewTask task)
            throws GeneralException {

        List items = task.getItems();

        for (Object obj : items) {

            ReviewItem item = (ReviewItem) obj;

            if ("Revoke".equals(item.getDecision())) {

                revokeAccess(item);

            }

        }

    }

    private void revokeAccess(ReviewItem item)
            throws GeneralException {

        System.out.println(
                "Revoking Access for Identity: "
                        + item.getIdentityName()
                        + " Role: "
                        + item.getTargetName()
        );

    }
}

import sailpoint.api.SailPointContext;
import sailpoint.object.Campaign;
import sailpoint.object.ReviewTask;
import sailpoint.object.ReviewItem;
import sailpoint.tools.GeneralException;

import java.util.List;

public class CertificationSignOffProcessor {

    private SailPointContext context;

    public CertificationSignOffProcessor(SailPointContext context) {
        this.context = context;
    }

    public void processSignOff(String campaignName)
            throws GeneralException {

        Campaign campaign =
                context.getObjectByName(Campaign.class, campaignName);

        if (campaign == null) {
            System.out.println("Campaign not found");
            return;
        }

        System.out.println("Campaign Loaded: " + campaignName);

        List tasks = campaign.getTasks();

        for (Object obj : tasks) {

            ReviewTask task = (ReviewTask) obj;

            processTask(task);
        }

        campaign.setStatus("Completed");

        context.saveObject(campaign);

        context.commitTransaction();

        System.out.println("Campaign Signed Off Successfully");
    }

    private void processTask(ReviewTask task)
            throws GeneralException {

        System.out.println("Processing Task: " + task.getName());

        List items = task.getItems();

        for (Object obj : items) {

            ReviewItem item = (ReviewItem) obj;

            processItem(item);
        }
    }

    private void processItem(ReviewItem item)
            throws GeneralException {

        String decision = item.getDecision();

        System.out.println(
                "Current Decision for Identity "
                        + item.getIdentityName()
                        + " Role "
                        + item.getTargetName()
                        + " = "
                        + decision
        );

        if ("Approve".equals(decision)) {

            item.setDecision("Revoke");

            context.saveObject(item);

            System.out.println(
                    "Decision changed to Revoke for Identity: "
                            + item.getIdentityName()
                            + " Role: "
                            + item.getTargetName()
            );

            revokeAccess(item);
        }
    }

    private void revokeAccess(ReviewItem item)
            throws GeneralException {

        System.out.println(
                "Revoking Access for Identity: "
                        + item.getIdentityName()
                        + " Role: "
                        + item.getTargetName()
        );

    }
}
