4️⃣ Data Governance Methods:-
1️⃣ classifyData(dataObject, sensitivityLevel):-
Kya karta hai?
Ye method data ko classify karta hai ki wo kitna sensitive hai.
Common sensitivity levels:-
Public → sab log access kar sakte hain
Internal → company ke employees ke liye
Confidential → limited users
Highly Confidential → sirf special authorized users.

Example:-
classifyData("Salary.xlsx", "Confidential");
Expected Output:-
Data Classified Successfully
File: Salary.xlsx
Sensitivity Level: Confidential
Real Scenario:-
HR department ke salary files ko Confidential mark kiya jata hai taaki sab employees access na kar saken.

2️⃣ assignDataAccess(user, dataObject):-
Kya karta hai?
Ye method specific user ko specific data ka access deta hai.
Example:-
assignDataAccess("Rahul", "Salary.xlsx");
Expected Output
Access Granted
User: Rahul
Data: Salary.xlsx
Permission: Read
Real Scenario:-
HR manager ko employee salary data ka access diya jata hai.

3️⃣ auditDataAccess(user, dataObject):-
Kya karta hai?
Ye method check karta hai ki kis user ne data kab access kiya.
Isse audit trail generate hota hai.
Example:-
auditDataAccess("Rahul", "Salary.xlsx");
Expected Output
Audit Log
User: Rahul
Data Accessed: Salary.xlsx
Access Time: 15-Mar-2026 11:15 AM
Action: View
Real Scenario:-
Agar kisi ne confidential document open kiya, system uska record store karta hai.

4️⃣ removeDataAccess(user, dataObject):-
Kya karta hai?
Ye method user ka data access remove kar deta hai.
Example:-
removeDataAccess("Rahul", "Salary.xlsx");
Expected Output
Access Revoked
User: Rahul
Data: Salary.xlsx
Real Scenario:-
Employee department change kare ya resign kare to sensitive data access remove kar diya jata hai.

Java IAM Governance System (Without Lambda)
import java.util.*;

public class IAMGovernanceSystem {

    static Map<String, String> identities = new HashMap<String, String>();
    static Map<String, List<String>> userRoles = new HashMap<String, List<String>>();
    static Map<String, List<String>> userAccess = new HashMap<String, List<String>>();

    // ---------------- Identity Governance ----------------

    public static void createIdentity(String name){
        identities.put(name,"Active");
        System.out.println("Identity created: "+name);
    }

    public static void updateIdentity(String name,String dept){
        System.out.println("Identity updated: "+name+" department -> "+dept);
    }

    public static void disableIdentity(String name){
        identities.put(name,"Disabled");
        System.out.println("Identity disabled: "+name);
    }

    public static void enableIdentity(String name){
        identities.put(name,"Active");
        System.out.println("Identity enabled: "+name);
    }

    public static void getIdentity(String name){
        System.out.println("Identity Status: "+name+" -> "+identities.get(name));
    }

    public static void deleteIdentity(String name){
        identities.remove(name);
        System.out.println("Identity deleted: "+name);
    }


    // ---------------- Access Governance ----------------

    public static void grantAccess(String user,String resource,String entitlement){

        if(!userAccess.containsKey(user)){
            userAccess.put(user,new ArrayList<String>());
        }

        userAccess.get(user).add(resource+"-"+entitlement);

        System.out.println("Access Granted: "+user+" -> "+resource+" ("+entitlement+")");
    }

    public static void revokeAccess(String user,String resource){

        if(userAccess.containsKey(user)){

            List<String> accessList = userAccess.get(user);

            Iterator<String> iterator = accessList.iterator();

            while(iterator.hasNext()){
                String access = iterator.next();

                if(access.startsWith(resource)){
                    iterator.remove();
                }
            }
        }

        System.out.println("Access Revoked: "+user+" -> "+resource);
    }

    public static void requestAccess(String user,String resource){
        System.out.println("Access Request Created: "+user+" -> "+resource);
    }

    public static void approveAccess(String request){
        System.out.println("Access Request Approved: "+request);
    }


    // ---------------- Privileged Access Governance ----------------

    public static void assignPrivilegedRole(String user,String role){
        System.out.println("Privileged Role Assigned: "+user+" -> "+role);
    }

    public static void revokePrivilegedRole(String user,String role){
        System.out.println("Privileged Role Revoked: "+user+" -> "+role);
    }

    public static void auditPrivilegedAccess(){
        System.out.println("Privileged Access Audit Running...");
    }

    public static void monitorPrivilegedActivity(){
        System.out.println("Monitoring Admin Activities...");
    }


    // ---------------- Data Governance ----------------

    public static void classifyData(String data,String level){
        System.out.println("Data Classified: "+data+" -> "+level);
    }

    public static void assignDataAccess(String user,String data){
        System.out.println("Data Access Granted: "+user+" -> "+data);
    }

    public static void auditDataAccess(String user,String data){
        System.out.println("Audit Log: "+user+" accessed "+data);
    }

    public static void removeDataAccess(String user,String data){
        System.out.println("Data Access Revoked: "+user+" -> "+data);
    }


    // ---------------- Risk Governance ----------------

    public static void analyzeRisk(){
        System.out.println("Risk Analysis Completed");
    }

    public static void detectSoDViolations(){
        System.out.println("SoD Violation Detected");
    }

    public static void generateRiskReport(){
        System.out.println("Risk Report Generated");
    }

    public static void remediateRisk(String user,String access){
        System.out.println("Risk Fixed: "+user+" -> "+access+" removed");
    }


    // ---------------- Account Methods ----------------

    public static void createAccount(String app){
        System.out.println("Account Created in "+app);
    }

    public static void updateAccount(String app){
        System.out.println("Account Updated in "+app);
    }

    public static void deleteAccount(String app){
        System.out.println("Account Deleted in "+app);
    }

    public static void disableAccount(String app){
        System.out.println("Account Disabled in "+app);
    }

    public static void getAccount(String app){
        System.out.println("Fetching Account Details from "+app);
    }


    // ---------------- Role Methods ----------------

    public static void assignRole(String user,String role){

        if(!userRoles.containsKey(user)){
            userRoles.put(user,new ArrayList<String>());
        }

        userRoles.get(user).add(role);

        System.out.println("Role Assigned: "+user+" -> "+role);
    }

    public static void removeRole(String user,String role){

        if(userRoles.containsKey(user)){
            userRoles.get(user).remove(role);
        }

        System.out.println("Role Removed: "+user+" -> "+role);
    }

    public static void getRoles(String user){
        System.out.println("Roles for "+user+" : "+userRoles.get(user));
    }

    public static void updateRole(String role){
        System.out.println("Role Updated: "+role);

        System.out.println("Manager Reviewing Certification");

SaaS Governance Java Project (Without Lambda & Without * Import)
import java.util.HashMap;
import java.util.ArrayList;
import java.util.List;
import java.util.Map;

public class SaaSGovernanceSystem {

    static Map<String,String> users = new HashMap<String,String>();
    static Map<String,List<String>> userApps = new HashMap<String,List<String>>();
    static List<String> auditLogs = new ArrayList<String>();


    // ---------------- USER MANAGEMENT ----------------

    public static void createUser(String name){

        users.put(name,"Active");

        System.out.println("User created: " + name);

        auditLogs.add("User created -> " + name);
    }

    public static void disableUser(String name){

        users.put(name,"Disabled");

        System.out.println("User disabled: " + name);

        auditLogs.add("User disabled -> " + name);
    }

    public static void getUserStatus(String name){

        System.out.println("User Status: " + users.get(name));
    }



    // ---------------- SAAS APPLICATION ----------------

    public static void addApplication(String user,String app){

        if(!userApps.containsKey(user)){

            userApps.put(user,new ArrayList<String>());
        }

        userApps.get(user).add(app);

        System.out.println("Application Added: " + user + " -> " + app);

        auditLogs.add("App access granted -> " + user + " -> " + app);
    }

    public static void removeApplication(String user,String app){

        if(userApps.containsKey(user)){

            List<String> list = userApps.get(user);

            int i;

            for(i=0;i<list.size();i++){

                if(list.get(i).equals(app)){

                    list.remove(i);

                    break;
                }
            }
        }

        System.out.println("Application Removed: " + user + " -> " + app);

        auditLogs.add("App access removed -> " + user + " -> " + app);
    }



    // ---------------- ACCESS REQUEST ----------------

    public static void requestAccess(String user,String app){

        System.out.println("Access request created: " + user + " -> " + app);

        auditLogs.add("Access request -> " + user + " -> " + app);
    }

    public static void approveAccess(String user,String app){

        addApplication(user,app);

        System.out.println("Manager approved access");

        auditLogs.add("Access approved -> " + user + " -> " + app);
    }



    // ---------------- PROVISIONING ----------------

    public static void provisionAccount(String user,String app){

        System.out.println("Provisioning account in " + app + " for " + user);

        auditLogs.add("Provisioning -> " + user + " -> " + app);
    }

    public static void deprovisionAccount(String user,String app){

        removeApplication(user,app);

        System.out.println("Deprovisioning account from " + app);

        auditLogs.add("Deprovision -> " + user + " -> " + app);
    }



    // ---------------- AUDIT ----------------

    public static void showAuditLogs(){

        System.out.println("------ AUDIT TRAIL ------");

        int i;

        for(i=0;i<auditLogs.size();i++){

            System.out.println(auditLogs.get(i));
        }
    }



    // ---------------- MAIN METHOD ----------------

    public static void main(String[] args) {

        createUser("Rahul");

        requestAccess("Rahul","Slack");

        approveAccess("Rahul","Slack");

        provisionAccount("Rahul","Slack");

        requestAccess("Rahul","Salesforce");

        approveAccess("Rahul","Salesforce");

        provisionAccount("Rahul","Salesforce");

        showAuditLogs();
    }
}
