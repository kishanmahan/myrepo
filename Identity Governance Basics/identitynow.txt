1️⃣ Identity:-
Identity = User / Employee
Example:-
Rahul
Amit
Priya
Identity ke attributes ho sakte hain:-
firstname
lastname
email
department
manager

Example:-
Name: Rahul
Department: IT

2️⃣ Source:-
Source = System jahan se user data aata hai
Examples:-
Active Directory
HR system
Database
Salesforce

Flow:-
HR System
   ↓
IdentityNow Source
   ↓
Identity create

3️⃣ Aggregation:-
Aggregation = Source se users aur accounts import karna
Example:-
Active Directory
      ↓
Aggregation
      ↓
IdentityNow me users load

4️⃣ Identity Profile:-
Identity Profile = Identity create karne ka rule
Ye decide karta hai:-
kaun sa source use hoga
kaun se attributes identity me aayenge
Example mapping:-
HR.firstName → Identity.firstName
HR.email → Identity.email

5️⃣ Transform:-
Transform = Data modify karna
Example:-
firstname + lastname → displayName
Example output:-
Rahul + Kumar = Rahul Kumar

6️⃣ Role:-
Role = Access ka group
Example:-
Finance Role
IT Admin Role
HR Role
Role me multiple permissions hote hain.

Example:-
Finance Role
   ↓
SAP access
Database access

7️⃣ Access Profile:-
Access Profile = Application access ka package
Example:-
Application: SAP
Access Profile: SAP Finance Access

8️⃣ Entitlement:-
Entitlement = Specific permission
Example:-
SAP_READ
SAP_WRITE
ADMIN_ACCESS

9️⃣ Lifecycle State:-
Lifecycle State = User ka status
Example:-
Joiner
Active
Leave

Flow:-

      ↓
Joiner State
      ↓
Access assign

🔟 Provisioning:-

public class Identity {

    private String name;
    private String department;
    private String email;

    public Identity(String name, String department, String email) {
        this.name = name;
        this.department = department;
        this.email = email;
    }

    public String getName() {
        return name;
    }

    public String getDepartment() {
        return department;
    }

    public String getEmail() {
        return email;
    }
}

4️⃣ AccessProfile.java
import java.util.ArrayList;
import java.util.List;

public class AccessProfile {

    private String name;
    private List<Entitlement> entitlements = new ArrayList<>();

    public AccessProfile(String name) {
        this.name = name;
    }

    public void addEntitlement(Entitlement e) {
        entitlements.add(e);
    }

    public String getName() {
        return name;
    }

    public List<Entitlement> getEntitlements() {
        return entitlements;
    }
}

7️⃣ ProvisioningService.java
public class ProvisioningService {

    public void provisionAccess(AccessRequest request) {

        System.out.println("Provisioning access for: " + request.getUser().getName());

        for (Entitlement e : request.getProfile().getEntitlements()) {
            System.out.println("Granting permission: " + e.getPermission());
        }

        System.out.println("Access Granted Successfully");
    }
}
8️⃣ CertificationService.java
public class CertificationService {

    public void reviewAccess(Identity user) {

        System.out.println("Manager reviewing access for: " + user.getName());
        System.out.println("Decision: Access Approved");
    }
}
9️⃣ Main.java (Complete Flow)
public class Main {

    public static void main(String[] args) {

        // Source → HR System user
        Identity rahul = new Identity("Rahul", "Finance", "rahul@company.com");

        // Entitlement
        Entitlement read = new Entitlement("SAP_READ");
        Entitlement write = new Entitlement("SAP_WRITE");

        // Access Profile
        AccessProfile sapFinance = new AccessProfile("SAP Finance Access");
        sapFinance.addEntitlement(read);
        sapFinance.addEntitlement(write);

        // Role Assignment
        Role financeRole = new Role("Finance Role");
        financeRole.addAccessProfile(sapFinance);

        // Access Request
        AccessRequest request = new AccessRequest(rahul, sapFinance);

        System.out.println("User " + rahul.getName() + " requested " + sapFinance.getName());

        // Approval Workflow
        CertificationService certification = new CertificationService();
        certification.reviewAccess(rahul);

        // Provisioning
        ProvisioningService provisioning = new ProvisioningService();
        provisioning.provisionAccess(request);

        System.out.println("Governance Audit Completed");
    }
}
