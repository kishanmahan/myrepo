    PreparedStatement stmt = null;
    logs.error("disableAccount method called "); // Retained image log as is
    ProvisioningResult result = new ProvisioningResult();
    String enableQueryAuth = "Update OSPAuth..[user] set userStatus = ? where login = ?";
    
    try {
        stmt = connection.prepareStatement(enableQueryAuth);
        stmt.setBoolean(1, true);
        stmt.setString(2, name);
        int userStatus1 = stmt.executeUpdate();
        logs.debug("OSPAuth db enable successfully ");
        
        if (userStatus1 > 0) {
            try {
                if (stmt != null) stmt.close();
            } catch (Exception e) {
                logs.error("Error closing statement", e);
            }
            
            String enableQueryAdmin = "Update OSPAdmin..[user] set status = ? where name = ?";
            stmt = connection.prepareStatement(enableQueryAdmin);
            stmt.setBoolean(1, true);
            stmt.setString(2, name);
            int userStatus2 = stmt.executeUpdate();
            
            if (userStatus2 > 0) {
                result.setStatus(ProvisioningResult.STATUS_COMMITTED);
                logs.debug("OSPAdmin db enable successfully ");
                return result;
            }
        }
    } catch (Exception e) {
        result.setStatus(ProvisioningResult.STATUS_FAILED);
        logs.error("update OSPAdmin error ", e);
        return result;
    } else {
        result.setStatus(ProvisioningResult.STATUS_FAILED);
    }
    
    // Sudhar: Finally block block resource leaks se bachane ke liye
    finally {
        try { if (stmt != null) stmt.close(); } catch (Exception e) { logs.error("Error closing statement", e); }
    }
    return result;
}

ProvisioningResult deleteAccount(String name) {
    PreparedStatement stmt = null;
    logs.debug("deleteAccount method called ");
    ProvisioningResult result = new ProvisioningResult();
    
    String deleteQuery = "BEGIN " +
        "DECLARE @Userid UNIQUEIDENTIFIER; " +
        "SELECT @Userid = userId FROM OSPAuth..[user] WHERE login = ?; " +
        "DELETE OSPAuth..access_token FROM OSPAuth..access_token at " +
        "JOIN OSPAuth..auth a ON a.authId = at.refreshId WHERE a.userId = @Userid; " +
        "DELETE OSPAuth..refresh_token FROM OSPAuth..refresh_token rt " +
        "JOIN OSPAuth..auth a ON a.authId = rt.refreshId WHERE a.userId = @Userid; " +
        "DELETE FROM OSPAuth..auth WHERE userId = @Userid; " +
        "DELETE FROM OSPAdmin..settings WHERE userId = @Userid; " +

        "DELETE FROM OSPAuth..[user] WHERE userId = @Userid; " +

    ResultSet rs = null;
    String per = null;
    String roleQuery = "SELECT * FROM OSPAdmin..[user] WHERE name = ? ";
    
    try {

// Helper Method: Add Permissions String Creation
int addPermission(String name, List permissionList) {
    PreparedStatement stmt = null;
    String addPermissionQuery = "Update OSPAdmin..[user] set permissions = ? where name = ?";
    StringBuilder sb = new StringBuilder();
    String entitlementString = null;
    
    sb.append("[ { \"action\" : \"basic\" },");
    for (String ent : permissionList) {
        if ("view".equalsIgnoreCase(ent)) {
            sb.append("{ \"action\" : \"view\" },");
        }
        if ("edit".equalsIgnoreCase(ent)) {
            sb.append("{ \"action\" : \"edit\" },");
        }
        if ("admin".equalsIgnoreCase(ent)) {
            sb.append("{ \"action\" : \"admin\" },");
        }
    }
    
    // Sudhar: Trailing comma hatane ka fix taaki JSON syntax query break na kare
    if (sb.charAt(sb.length() - 1) == ',') {
        sb.deleteCharAt(sb.length() - 1);
    }
    sb.append(" ]");
    
    entitlementString = sb.toString();
    logs.error("entitlementString::::" + entitlementString);
    
    try {
        stmt = connection.prepareStatement(addPermissionQuery);
        stmt.setString(1, entitlementString);
        stmt.setString(2, name);
        int executeStatus = stmt.executeUpdate();
        return executeStatus;
    } catch (Exception e) {
        logs.error("Error executing permission update", e);
        return 0;
    } finally {
        try { if (stmt != null) stmt.close(); } catch (Exception e) {}
    }
}

// Helper Method: Modify Account Block
ProvisioningResult modifyAccount(AccountRequest acctReq) {
    ProvisioningResult result = new ProvisioningResult();
    List permissionAddList = new LinkedList();
    List permissionRemoveList = new LinkedList();
    List existPer = new ArrayList();
    Set finalPer = new HashSet();
    
    String nativeIdentity = acctReq.getNativeIdentity();
    List attributeRequests = acctReq.getAttributeRequests();
    
    for (AttributeRequest ar : attributeRequests) {
        if ("permissions".equalsIgnoreCase(ar.getName())) {
            if ("Add".equalsIgnoreCase(ar.getOperation().toString()) || "Set".equalsIgnoreCase(ar.getOperation().toString())) {
                if (ar.getValue() instanceof String) {
                    permissionAddList.add((String) ar.getValue());
                } else if (ar.getValue() instanceof List) {
                    permissionAddList = (List) ar.getValue();
                }
            } else if ("Remove".equalsIgnoreCase(ar.getOperation().toString())) {
                if (ar.getValue() instanceof String) {
                    permissionRemoveList.add((String) ar.getValue());
                } else if (ar.getValue() instanceof List) {
                    permissionRemoveList = (List) ar.getValue();
                }
            }
        }
    }
    
    String per = getRole(nativeIdentity);
    logs.error("permissionList:::;" + permissionAddList);
    
    if (per != null) {
        Pattern pattern = Pattern.compile("action\\s*:\\s*\\s*(['\"]+[^'\"]+)");
        Matcher matcher = pattern.matcher(per);
        while (matcher.find()) {
            existPer.add(matcher.group(1));
        }
    }
    
    finalPer.addAll(existPer);
    finalPer.addAll(permissionAddList);
    finalPer.removeAll(permissionRemoveList);
    finalPer.remove("basic");
    
    logs.error("final permissions;;;;" + finalPer);
    
    int addEntStatus = addPermission(nativeIdentity, new ArrayList(finalPer));
    if (addEntStatus > 0) {
        result.setStatus(ProvisioningResult.STATUS_COMMITTED);
    } else {
        result.setStatus(ProvisioningResult.STATUS_FAILED);
    }
    return result;
}

                stmt.close();
            } catch (Exception e) {}

            stmt = connection.prepareStatement(disableQueryAdmin);
            stmt.setBoolean(1, false);
            stmt.setString(2, name);
            int userStatus2 = stmt.executeUpdate();
            if (userStatus2 > 0) {
                result.setStatus(ProvisioningResult.STATUS_COMMITTED);
                logs.error("OSPAdmin db Disable successfully ");
                return result;
            }
            stmt.close();
        }
    } catch (Exception e) {
        result.setStatus(ProvisioningResult.STATUS_FAILED);
        logs.error("update OSPAdmin error ", e);
        return result;
    } else {
        result.setStatus(ProvisioningResult.STATUS_FAILED);
    }
    try {
        if (stmt != null) {
            stmt.close();
        }
    } catch (Exception e) {
        logs.error("Error closing statement ", e);
    }
    return result;
}

ProvisioningResult modifyAccount(AccountRequest acctReq) {
    PreparedStatement stmt = null;
    logs.error("modifyAccount method called ");
    ProvisioningResult result = new ProvisioningResult();
    List<String> permissionAddList = new LinkedList<String>();
    List<String> permissionRemoveList = new LinkedList<String>();
    List<String> existPer = new ArrayList<String>();
    Set finalPer = new HashSet();
    String nativeIdentity = acctReq.getNativeIdentity();
    List<AttributeRequest> attributeRequests = acctReq.getAttributeRequests();
    
    for (AttributeRequest ar : attributeRequests) {
        if ("permissions".equalsIgnoreCase(ar.getName())) {
            if ("Add".equalsIgnoreCase(ar.getOperation().toString()) || "Set".equalsIgnoreCase(ar.getOperation().toString())) {
                if (ar.getValue() instanceof String) {
                    permissionAddList.add((String) ar.getValue());
                } else if (ar.getValue() instanceof List) {
                    permissionAddList = (List) ar.getValue();
                }
            } else if ("Remove".equalsIgnoreCase(ar.getOperation().toString())) {
                if (ar.getValue() instanceof String) {
                    permissionRemoveList.add((String) ar.getValue());
                } else if (ar.getValue() instanceof List) {
                    permissionRemoveList = (List) ar.getValue();
                }
            }
        }
    }
    
    String per = getRole(nativeIdentity);
    logs.debug("existing permissions;;;" + per);
    logs.debug("permissionAddList;;;" + permissionAddList);
    logs.error("permissionRemoveList;;;" + permissionRemoveList);
    
    if (per != null) {
        Pattern pattern = Pattern.compile("action\\s*:\\s*\\s*\"([^\"]+)\"");
        Matcher matcher = pattern.matcher(per);
        while (matcher.find()) {
            existPer.add(matcher.group(1));
        }
    }
    
    finalPer.addAll(existPer);
    finalPer.addAll(permissionAddList);
    finalPer.removeAll(permissionRemoveList);
    finalPer.remove("basic");
    logs.error("final permissions;;;;" + finalPer);
    
    int addEntStatus = addPermission(nativeIdentity, new ArrayList(finalPer));
    if (addEntStatus > 0) {
        result.setStatus(ProvisioningResult.STATUS_COMMITTED);
    } else {
        result.setStatus(ProvisioningResult.STATUS_FAILED);
    }
    return result;
}
