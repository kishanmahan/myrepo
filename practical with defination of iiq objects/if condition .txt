import sailpoint.object.*;
import java.util.*;

// 1. Apni application ka sahi naam yahan likhein
String targetAppName = "identity_data_90_lines"; 
int count = 0;

// 2. QueryOptions banayein jo sirf un identities ko uthaye jinke paas is application ka account hai
QueryOptions qo = new QueryOptions();
qo.addFilter(Filter.eq("links.application.name", targetAppName));
// spadmin ko chhodne ke liye
qo.addFilter(Filter.ne("name", "spadmin")); 

Iterator it = context.search(Identity.class, qo);

while (it.hasNext()) {
    Identity identity = (Identity) it.next();
    
    // Password set karein
    identity.setPassword("Welcome@123");
    
    // Employee capability add karein (optional, jaisa aapne pehle maanga tha)
    Capability cap = context.getObjectByName(Capability.class, "Employee");
    if(cap != null && !identity.hasCapability(cap.getName())) {
        identity.addCapability(cap);
    }

    context.saveObject(identity);
    count++;
    
    // Memory management ke liye (agar users zyada hain)
    if (count % 50 == 0) {
        context.commitTransaction();
        context.decache();
    }
}

context.commitTransaction();
return "Total " + count + " identities from " + targetAppName + " updated successfully!";
