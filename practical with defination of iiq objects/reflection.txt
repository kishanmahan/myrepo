ReflectionDemo.java
Java
Copy code
package com.test;

import java.lang.reflect.Constructor;
import java.lang.reflect.Modifier;

public class ReflectionDemo {

    public static void main(String[] args) {

        try {

            // Class Object
            Class<Student> cls = Student.class;

            System.out.println("Class Name : "
                    + cls.getName());

            System.out.println("\n===== ALL CONSTRUCTORS =====");

            Constructor<?>[] constructors =
                    cls.getDeclaredConstructors();

            for (Constructor<?> c : constructors) {

                System.out.println("Constructor Name : "
                        + c.getName());

                System.out.println("Parameter Count : "
                        + c.getParameterCount());

                System.out.println("Modifiers : "
                        + Modifier.toString(c.getModifiers()));

                Class<?>[] params =
                        c.getParameterTypes();

                System.out.print("Parameters : ");

                for (Class<?> p : params) {
                    System.out.print(
                            p.getSimpleName() + " ");
                }

                System.out.println();
                System.out.println("-------------------");
            }

            System.out.println("\n===== DEFAULT CONSTRUCTOR =====");

            Constructor<Student> c1 =
                    cls.getConstructor();

            Student s1 =
                    c1.newInstance();

            s1.display();

            System.out.println("\n===== PARAMETERIZED CONSTRUCTOR =====");

            Constructor<Student> c2 =
                    cls.getConstructor(
                            int.class,
                            String.class);

            Student s2 =
                    c2.newInstance(
                            101,
                            "Rahul");

            s2.display();

            System.out.println("\n===== PRIVATE CONSTRUCTOR =====");

            Constructor<Student> c3 =
                    cls.getDeclaredConstructor(
                            String.class);

            c3.setAccessible(true);

            Student s3 =
                    c3.newInstance(
                            "Aman");

            s3.display();

        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
