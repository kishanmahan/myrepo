import sailpoint.api.SailPointFactory;
import sailpoint.api.SailPointContext;
import sailpoint.object.Identity;
import sailpoint.object.ManagedAttribute;

public class RollbackExample {

    public static void main(String[] args) {

        SailPointContext context = null;

        try {
            context = SailPointFactory.createContext();

            Identity identity = context.getObjectByName(Identity.class, "Rahul");

            ManagedAttribute attr = context.getObjectByName(
                ManagedAttribute.class, "Dev Group"
            );

            identity.addAttribute("memberOf", attr.getValue());

            context.saveObject(identity);
            context.commitTransaction();

            System.out.println("Success: Access added");

        } catch (Exception e) {
            System.out.println("Error aaya: " + e.getMessage());

            if (context != null) {
                context.rollbackTransaction();
                System.out.println("Rollback ho gaya");
            }
			finally{
			if(context!=null)
			{
			context.close();
			}
        }
    }
}
