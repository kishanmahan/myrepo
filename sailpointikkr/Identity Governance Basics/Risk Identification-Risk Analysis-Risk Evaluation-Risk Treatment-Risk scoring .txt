()Java Project: Risk Management with Risk Scoring:-
import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

// Risk class
class Risk {
    private String name;
    private double probability; // 0.0 to 1.0
    private double impact;      // in currency
    private double score;
    private String priority;

    // Constructor
    public Risk(String name, double probability, double impact) {
        this.name = name;
        this.probability = probability;
        this.impact = impact;
        calculateScore();
        assignPriority();
    }

    // Score calculation
    private void calculateScore() {
        this.score = this.probability * this.impact;
    }

    // Priority assignment
    private void assignPriority() {
        if (score >= 75000) {
            this.priority = "High";
        } else if (score >= 25000) {
            this.priority = "Medium";
        } else {
            this.priority = "Low";
        }
    }

    // Display Risk
    public void displayRisk() {
        System.out.println("Risk Name   : " + name);
        System.out.println("Probability : " + probability);
        System.out.println("Impact      : " + impact);
        System.out.println("Risk Score  : " + score);
        System.out.println("Priority    : " + priority);
        System.out.println("--------------------------");
    }
}

// Main class
public class RiskManagementProject {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        List<Risk> riskList = new ArrayList<>();

        System.out.println("=== Risk Management & Scoring Project ===");

        while (true) {
            System.out.println("\n1. Add Risk");
            System.out.println("2. Show All Risks");
            System.out.println("3. Exit");
            System.out.print("Choose option: ");
            int choice = sc.nextInt();
            sc.nextLine(); // Consume newline

            if (choice == 1) {
                System.out.print("Enter Risk Name: ");
                String name = sc.nextLine();

                System.out.print("Enter Probability (0.0 - 1.0): ");
                double prob = sc.nextDouble();

                System.out.print("Enter Impact (в‚№): ");
                double impact = sc.nextDouble();

                Risk r = new Risk(name, prob, impact);
                riskList.add(r);
                System.out.println("вњ… Risk added successfully!");

            } else if (choice == 2) {
                if (riskList.isEmpty()) {
                    System.out.println("No risks added yet.");
                } else {
                    System.out.println("\n--- All Risks ---");
                    for (Risk r : riskList) {
                        r.displayRisk();
                    }
                }
            } else if (choice == 3) {
                System.out.println("Exiting... Bye!");
                break;
            } else {
                System.out.println("Invalid choice! Try again.");
            }
        }

        sc.close();
    }
}
