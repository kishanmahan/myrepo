--🔹 DelimitedFileConnector kya hai?
SailPoint IdentityIQ me DelimitedFileConnector ek file-based connector hai jo CSV / delimited files (jaise comma, pipe |, tab separated) ko read karke system me data import/export karta hai.

👉 Simple language me:-
“File (CSV etc.) se data read karke SailPoint me use karna”

🔹 Ye kya karta hai?
1️⃣ File se data read karta hai:-
Example file:-
sAMAccountName,displayName,department
kumar123,Kumar Singh,IT
rahul456,Rahul Verma,HR
👉 Connector is file ko read karega

2️⃣ Data ko Identity / Account me convert karta hai
File ka data:-
Identity
Account
Attribute
me convert ho jata hai

3️⃣ Aggregation (data fetch) karta hai:-
👉 Ye system se data “pull” karta hai:-
Accounts list
Attributes
Users

4️⃣ Provisioning ke liye data generate karta hai?
👉 File me jo data hai usko use karke:-
Create
Update
Delete
operations possible hote hain

🔹 Kya ye DB me store hota hai?

👉 ❌ Directly nahi
DelimitedFileConnector ka data database me store nahi hota
Ye external file source hota hai.

👉 Lekin:-
✔ Aggregated data IdentityIQ ke database me store ho sakta hai
✔ Connector ka configuration DB me hota hai.

👉 SailPoint IdentityIQ ke DelimitedFileConnector me
iterateObjects() ek core method hai jo file ke har record (row) par loop chalata hai.

🔹 Simple Meaning:-
👉 Ye method:-
CSV / delimited file ko row by row read karta hai
Har row ko ek object (map / bean) me convert karta hai
Aur us object par custom logic execute karwata hai.
import java.util.List;
import java.util.Map;
import java.util.ArrayList;
import java.io.BufferedReader;
import java.io.FileReader;
import java.io.IOException;

import sailpoint.object.ObjectHandler;

public class DelimitedFileExample {

    public static void main(String[] args) {

        List users = readFile("/data/users.csv");

        iterateObjects(users, new ObjectHandler() {

            public void handle(Object obj) {

                Map row = (Map) obj;

                String empId = (String) row.get("empId");
                String name  = (String) row.get("name");
                String dept  = (String) row.get("dept");

                System.out.println("EmpId: " + empId);
                System.out.println("Name: " + name);
                System.out.println("Dept: " + dept);
                System.out.println("------------------");
            }
        });
    }

    // 🔹 File read + convert into List<Map>
    public static List readFile(String filePath) {

        List list = new ArrayList();

        BufferedReader br = null;

        try {

            br = new BufferedReader(new FileReader(filePath));

            String line;

            // First line = header
            String headerLine = br.readLine();
            String[] headers = headerLine.split(",");

            while ((line = br.readLine()) != null) {

                String[] values = line.split(",");

                Map row = new java.util.HashMap();

                for (int i = 0; i < headers.length; i++) {

                    row.put(headers[i], values[i]);
                }

                list.add(row);
            }

        } catch (IOException e) {

            System.out.println("Error reading file: " + e.getMessage());

        } finally {

            try {
                if (br != null) br.close();
            } catch (IOException e) {
                e.printStackTrace();
            }
        }

        return list;
    }

    // 🔹 iterateObjects implementation
    public static void iterateObjects(List list, ObjectHandler handler) {

        if (list == null || handler == null) return;

        for (Object obj : list) {

            handler.handle(obj);
        }
    }
}

Step 1: FileReader:-

Step 2: BufferedReader:-
new BufferedReader(...)
✔ FileReader ko wrap karta hai
✔ Data ko buffer me store karta hai
✔ Line-by-line reading allow karta hai
👉 Final:-
BufferedReader br = new BufferedReader(new FileReader(filePath));
✔ File ko efficient way me read karne ke liye.

import sailpoint.api.SailPointContext;
import sailpoint.api.SailPointContextFactory;
import sailpoint.connector.DelimitedFileConnector;
import java.util.*;

public class TestConnector {

    public static void main(String[] args) throws Exception {

        // 1️⃣ Context lena
        SailPointContext context = SailPointContextFactory.getCurrentContext();

        // 2️⃣ Connector initialize karna
        DelimitedFileConnector connector = new DelimitedFileConnector();

        // 3️⃣ Config banana (real me ConnectorConfig object hota hai)
        Map<String, Object> config = new HashMap<>();
        config.put("filePath", "/data/users.csv");

        connector.initialize(config);

        // 4️⃣ Query call
        List<Map<String, Object>> data = connector.query();

        // 5️⃣ Data process
        for (Map<String, Object> row : data) {

            // attributes build
            Map<String, Object> attrs = connector.buildAttributes(row);

            // object build
            Object obj = connector.buildObject(attrs);

            // normalize example
            String name = (String) attrs.get("name");
            String cleanName = connector.normalize(name);

            System.out.println("Original: " + name);
            System.out.println("Clean: " + cleanName);
            System.out.println("Object: " + obj);
            System.out.println("-------------------");
        }
    }
}

import sailpoint.api.SailPointContext;
import sailpoint.api.SailPointContextFactory;
import sailpoint.connector.DelimitedFileConnector;

import java.util.List;
import java.util.Map;

public class DelimitedFileExample {

    public static void main(String[] args) {

        try {

            // 1️⃣ SailPoint Context lena
            SailPointContext context = SailPointContextFactory.getCurrentContext();

            // 2️⃣ Connector object create karna
            DelimitedFileConnector connector = new DelimitedFileConnector();

            // 3️⃣ Connector configuration (real system me ConnectorConfig hota hai)
            java.util.Map<String, Object> config = new java.util.HashMap<String, Object>();
            config.put("filePath", "/data/users.csv");
            config.put("delimiter", ",");
            config.put("hasHeader", "true");

            // 4️⃣ Initialize connector
            connector.initialize(config);

            // 5️⃣ Query execute karna
            List<Map<String, Object>> result = connector.executeQuery("SELECT * FROM users");

            // 6️⃣ Result print karna
            for (Map<String, Object> row : result) {

                System.out.println("----- ROW START -----");

                for (Map.Entry<String, Object> entry : row.entrySet()) {
                    System.out.println(entry.getKey() + " : " + entry.getValue());
                }

                System.out.println("----- ROW END -----");
            }

            // 7️⃣ Context close (good practice)
            context.decache();

        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}

import sailpoint.api.SailPointContext;
import sailpoint.api.SailPointContextFactory;
import sailpoint.object.Identity;

import java.util.HashMap;
import java.util.List;
import java.util.ArrayList;
import java.util.Map;

public class SailPointFlowDemo {

    public static void main(String[] args) throws Exception {

        // 1️⃣ Context lena
        SailPointContext context = SailPointContextFactory.getCurrentContext();

        // 2️⃣ Load Data (real like DB fetch)
        List<Map<String, Object>> data = loadData(context);

        // 3️⃣ Process Request
        Map<String, Object> request = new HashMap<String, Object>();
        request.put("operation", "update");

        processRequest(context, request);

        // 4️⃣ Apply Delta
        Map<String, Object> oldData = data.get(0);
        Map<String, Object> newData = new HashMap<String, Object>();
        newData.put("name", "ram_updated");

        applyDelta(context, oldData, newData);

        // 5️⃣ Sync
        sync(context);

        // 6️⃣ Refresh
        refresh(context, oldData);

        // 7️⃣ Save Data
        saveData(context, oldData);

        context.decache();
    }

    // 🔹 handleRequest (entry style)
    public static void handleRequest(SailPointContext context, Map<String, Object> request) {

        String op = (String) request.get("operation");

        System.out.println("handleRequest -> " + op);

        if ("create".equals(op)) {
            System.out.println("Creating object...");
        }
    }

    // 🔹 processRequest
    public static void processRequest(SailPointContext context, Map<String, Object> request) {

        System.out.println("processRequest -> Processing request");

        handleRequest(context, request);
    }

    // 🔹 applyDelta
    public static void applyDelta(SailPointContext context,
                                 Map<String, Object> oldData,
                                 Map<String, Object> newData) {

        System.out.println("applyDelta -> Before: " + oldData);

        oldData.putAll(newData);

        System.out.println("applyDelta -> After: " + oldData);
    }

    // 🔹 sync
    public static void sync(SailPointContext context) {

        System.out.println("sync -> Aggregation / sync started");

        // real systems me external system sync hota hai
    }

    // 🔹 refresh
    public static void refresh(SailPointContext context, Map<String, Object> data) {

        System.out.println("refresh -> Reloading data");

        // example: identity refresh
        String name = (String) data.get("name");

        System.out.println("Refreshed object: " + name);
    }

    // 🔹 loadData (REAL API use)
    public static List<Map<String, Object>> loadData(SailPointContext context) throws Exception {

        List<Map<String, Object>> list = new ArrayList<Map<String, Object>>();

        Identity id = context.getObjectByName(Identity.class, "john.doe");

        Map<String, Object> data = new HashMap<String, Object>();
        data.put("name", id.getName());

        list.add(data);

        System.out.println("loadData -> Loaded: " + list);

        return list;
    }

    // 🔹 saveData (REAL API use)
    public static void saveData(SailPointContext context, Map<String, Object> data) throws Exception {

        String name = (String) data.get("name");

        Identity id = context.getObjectByName(Identity.class, name);

        id.setAttribute("status", "UPDATED");

        context.saveObject(id);
        context.commitTransaction();

        System.out.println("saveData -> Saved: " + name);
    }
}
