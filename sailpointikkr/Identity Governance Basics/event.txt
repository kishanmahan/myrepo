🚀 FULL REAL CODE (Rule + Workflow Trigger)
import sailpoint.api.SailPointContext;
import sailpoint.object.Identity;
import sailpoint.object.Workflow;
import java.util.Map;
import java.util.HashMap;

public class EventRule {

    public static void execute(SailPointContext context, String identityName) throws Exception {

        Identity identity = context.getObjectByName(Identity.class, identityName);

        if (identity != null) {

            // 🔥 1. Identity Created Event
            if (identity.isNew()) {
                System.out.println("Event: Identity Created");

                Map<String, Object> args = new HashMap<String, Object>();
                args.put("identityName", identity.getName());

                Workflow wf = context.getObjectByName(Workflow.class, "IdentityCreationWorkflow");
                context.startWorkflow(wf, args);
            }

            // 🔥 2. Identity Updated Event
            else if (identity.isModified()) {
                System.out.println("Event: Identity Updated");

                Map<String, Object> args = new HashMap<String, Object>();
                args.put("identityName", identity.getName());

                Workflow wf = context.getObjectByName(Workflow.class, "IdentityUpdateWorkflow");
                context.startWorkflow(wf, args);
            }

            // 🔥 3. Identity Deleted Event
            else if (identity.isDeleted()) {
                System.out.println("Event: Identity Deleted");

                Map<String, Object> args = new HashMap<String, Object>();
                args.put("identityName", identity.getName());

                Workflow wf = context.getObjectByName(Workflow.class, "IdentityDeleteWorkflow");
                context.startWorkflow(wf, args);
            }
        }
    }
}
🔥 Workflow XML (Real SailPoint Style)
<Workflow name="IdentityCreationWorkflow">
    <Step name="Start">
        <Transition to="Provision"/>
    </Step>

    <Step name="Provision">
        <Action>Provision Identity</Action>
        <Transition to="End"/>
    </Step>

    <Step name="End">
        <End/>
    </Step>
</Workflow>

🔹 <Transition to="Provision"/>:-
👉 Transition = next step kaha jayega
