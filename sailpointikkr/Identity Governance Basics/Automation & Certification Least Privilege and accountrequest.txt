import sailpoint.api.SailPointContext;
import sailpoint.object.Identity;
import sailpoint.object.Link;
import sailpoint.object.Application;
import sailpoint.object.ProvisioningPlan;
import sailpoint.object.ProvisioningPlan.AccountRequest;
import sailpoint.object.ProvisioningPlan.AttributeRequest;
import sailpoint.object.ProvisioningPlan.Operation;

import java.util.List;

public class LeastPrivilegeExample {

    public static void applyLeastPrivilege(SailPointContext context) throws Exception {

        // Identity fetch
        Identity user = context.getObjectByName(Identity.class, "john.doe");

        // Application fetch
        Application app = context.getObjectByName(Application.class, "ActiveDirectory");

        // User ke accounts
        List<Link> links = user.getLinks();

        // Provisioning Plan
        ProvisioningPlan plan = new ProvisioningPlan();
        plan.setIdentity(user);

        for(Link link : links){

            if(link.getApplicationName().equals("ActiveDirectory")){

                // Example: Remove extra entitlement
                AccountRequest accReq = new AccountRequest();
                accReq.setApplication(app.getName());
                accReq.setNativeIdentity(link.getNativeIdentity());
                accReq.setOperation(Operation.Modify);

                AttributeRequest attrReq = new AttributeRequest();
                attrReq.setName("memberOf");
                attrReq.setOperation(Operation.Remove);
                attrReq.setValue("AdminGroup");

                accReq.add(attrReq);

                plan.add(accReq);

            }

        }

        // Save provisioning plan
        context.saveObject(plan);
        context.commitTransaction();

        System.out.println("Least Privilege applied for user: " + user.getName());
    }
}
