()Privileged Access kya hai?
Ye access high-risk aur sensitive operations ke liye hota hai.
Normal users ke liye nahi, sirf admins, managers, aur IT/security personnel ke liye.
Purpose: critical systems me changes safe aur controlled ho.
Ek line me: “High-power access for trusted users”

2️⃣ Examples of Privileged Access:-
User	             Privileged Access	                    Notes
IT Admin-----------Server Admin, Network Config------Can change system settings
Finance Manager----Payroll approval------------------Can approve salary changes
HR Admin-----------Employee data write/delete--------Sensitive PII data

4️⃣ Privileged Access Management in SailPoint,
SailPoint me PAM (Privileged Access Management) ka part hota hai:-
Roles define karna for privileged users
Approval workflow for granting temporary privileged access
Monitoring & Certification to track misuse.

5️⃣ Code Conceptual Example:-
Identity anil = getIdentity("EMP54321");
Role itAdmin = new Role("IT Admin");

// Assign privileged permissions
itAdmin.addPermission("Server Admin");
itAdmin.addPermission("Network Config");

// Access check
if(anil.hasPermission("Server Admin")){
    System.out.println("Privileged Access Granted!");
}else{
    System.out.println("Access Denied!");
}
✅ Tip bhai:-
Privileged access risky hota hai, isliye temporary, monitored aur approved hona chahiye.
SailPoint me ye access least privilege + time-bound + certification workflows ke sath manage hota hai.

()Enterprise PAM Example (Java):-
import java.util.ArrayList;
import java.util.Date;
import java.util.List;

class Permission {

    String name;

    Permission(String name) {
        this.name = name;
    }
}

class PrivilegedRole {

    String roleName;
    List<Permission> permissions = new ArrayList<Permission>();

    PrivilegedRole(String roleName) {
        this.roleName = roleName;
    }

    void addPermission(Permission p) {
        permissions.add(p);
    }
}

class Identity {

    String name;
    boolean mfaVerified;

    Identity(String name) {
        this.name = name;
    }

    void verifyMFA() {
        mfaVerified = true;
        System.out.println("MFA verified for " + name);
    }
}

class PrivilegedAccessGrant {

    Identity user;
    PrivilegedRole role;
    Date start;
    Date end;

    PrivilegedAccessGrant(Identity user, PrivilegedRole role, Date start, Date end) {
        this.user = user;
        this.role = role;
        this.start = start;
        this.end = end;
    }

    void checkAccess() {

        Date today = new Date();

        if (!user.mfaVerified) {
            System.out.println("Access Denied: MFA not verified");
            return;
        }

        if (today.after(start) && today.before(end)) {

            System.out.println("Privileged Access Granted to " + user.name);

            for (Permission p : role.permissions) {
                System.out.println("Permission: " + p.name);
            }

        } else {
            System.out.println("Privileged Access Expired");
        }
    }
}

class AuditLogger {

    static void log(String action) {
        System.out.println("AUDIT: " + action);
    }
}

public class PAMExample {

    public static void main(String[] args) {

        Identity rahul = new Identity("Rahul");

        PrivilegedRole dbAdmin = new PrivilegedRole("Database Admin");

        dbAdmin.addPermission(new Permission("Read Database"));
        dbAdmin.addPermission(new Permission("Modify Database"));
        dbAdmin.addPermission(new Permission("Delete Database"));

        rahul.verifyMFA();

        Date start = new Date();

        Date end = new Date(System.currentTimeMillis() + (2 * 60 * 60 * 1000));
        // 2 hours access

        PrivilegedAccessGrant grant =
                new PrivilegedAccessGrant(rahul, dbAdmin, start, end);

        grant.checkAccess();

        AuditLogger.log("Privileged access used by Rahul");
    }
}

import java.util.Random;
import java.util.Scanner;
import java.io.OutputStream;
import java.net.HttpURLConnection;
import java.net.URL;

class Identity {

    String name;
    String mobileNumber;
    String systemOtp;
    boolean mfaVerified = false;

    Identity(String name, String mobileNumber) {
        this.name = name;
        this.mobileNumber = mobileNumber;
    }

    void sendOTP() {

        systemOtp = generateOTP();

        SmsService.sendSMS(mobileNumber, systemOtp);

        System.out.println("OTP sent to mobile number: " + mobileNumber);
    }

    void verifyMFA(Scanner scanner) {

        System.out.print("Enter OTP for " + name + ": ");
        String enteredOtp = scanner.nextLine();

        if(systemOtp.equals(enteredOtp)) {
            mfaVerified = true;
            System.out.println("MFA verified for " + name);
        } else {
            System.out.println("Invalid OTP for " + name);
        }
    }

    String generateOTP() {

        Random r = new Random();
        int otp = 100000 + r.nextInt(900000);

        return String.valueOf(otp);
    }
}

class SmsService {

    static void sendSMS(String mobile, String otp) {

        try {

            String apiKey = "YOUR_API_KEY";

            String message = "Your OTP is " + otp;

            String data = "authkey=" + apiKey +
                    "&mobiles=" + mobile +
                    "&message=" + message +
                    "&sender=OTPMSG" +
                    "&route=4";

            URL url = new URL("https://api.msg91.com/api/sendhttp.php");

            HttpURLConnection conn = (HttpURLConnection) url.openConnection();

            conn.setRequestMethod("POST");
            conn.setDoOutput(true);

            OutputStream os = conn.getOutputStream();
            os.write(data.getBytes());
            os.flush();
            os.close();

            int responseCode = conn.getResponseCode();

            System.out.println("SMS API Response Code: " + responseCode);

        } catch(Exception e) {
            e.printStackTrace();
        }
    }
}

public class MFAExample {

    public static void main(String[] args) {

        Scanner scanner = new Scanner(System.in);

        Identity rahul = new Identity("Rahul","9876543210");
        Identity ram = new Identity("Ram","9876543222");
        Identity raman = new Identity("Raman","9876543333");

        // Send OTP
        rahul.sendOTP();
        ram.sendOTP();
        raman.sendOTP();

        // User enters OTP
        rahul.verifyMFA(scanner);
        ram.verifyMFA(scanner);
        raman.verifyMFA(scanner);

        scanner.close();
    }
}
